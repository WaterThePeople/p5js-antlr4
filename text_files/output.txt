
function setup() {
  createCanvas(600, 600);
  background(255);
    rekurencja(0, 0, 300, 0);
}

function kwadrat(x, y, len, rot) {
  push();
  translate(x, y);
  rotate(rot);
  fill(0,0,0,0);
  rect(0, 0, len, len);
  pop();
}

function trojkat(x, y, len, rot) {
  push();
  translate(x, y);
  rotate(rot);
  fill(0,0,0,0);
  triangle(
    0, 0,
    len / 2, -len * sqrt(3) / 2,
    len, 0
  );
  pop();
}

function rekurencja(x, y, len, rot) {
    if (len < 0.05) return;
    push();
    translate(x * len, y * len);
    rotate(rot);
    kwadrat(x + 0 * len, y + 0 * len, len * 1, 0);
  rekurencja(x + 0 * len, y + 0 * len, len * 0.5, 0);
    pop();
  }
