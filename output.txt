function tri(x, y, l) {
  triangle(x, y, (x + (l / 2)), (y - (((l * Math.sqrt(3))) / 2)), (x + l), y);
}

function sierpinski(x, y, scale, depth) {
  if (depth === 0) {
  tri(x, y, scale);
}
  if (depth > 0) {
  sierpinski(x, y, (scale / 2), (depth - 1));
  sierpinski((x + (scale / 2)), y, (scale / 2), (depth - 1));
  sierpinski((x + (scale / 4)), (y - (((scale * Math.sqrt(3))) / 4)), (scale / 2), (depth - 1));
}
}

sierpinski(200, 400, 400, 5);
